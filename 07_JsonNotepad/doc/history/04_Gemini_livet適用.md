The Model handles the logic for JSON serialization and file I/O. We use Newtonsoft.Json (ensure it's installed via NuGet) for robust JSON handling.

using System;
using System.IO;
using Newtonsoft.Json;
using Livet;

namespace JsonNotepad.Models
{
    public class NotepadData
    {
        public string Content { get; set; }
        public DateTime LastUpdated { get; set; }
    }

    public class JsonNotepadModel : NotificationObject
    {
        public NotepadData Load(string filePath)
        {
            if (!File.Exists(filePath)) return new NotepadData { Content = "" };

            string json = File.ReadAllText(filePath);
            return JsonConvert.DeserializeObject<NotepadData>(json);
        }

        public void Save(string filePath, string content)
        {
            var data = new NotepadData
            {
                Content = content,
                LastUpdated = DateTime.Now
            };
            string json = JsonConvert.SerializeObject(data, Formatting.Indented);
            File.WriteAllText(filePath, json);
        }
    }
}

using Livet;
using Livet.Commands;
using Livet.Messaging.IO;
using JsonNotepad.Models;

namespace JsonNotepad.ViewModels
{
    public class MainWindowViewModel : ViewModel
    {
        private JsonNotepadModel _model;

        public MainWindowViewModel()
        {
            _model = new JsonNotepadModel();
        }

        // --- Properties ---

        private string _textContent;
        public string TextContent
        {
            get => _textContent;
            set => RaisePropertyChangedIfActionChanged(ref _textContent, value);
        }

        // --- Commands ---

        private ViewModelCommand _saveCommand;
        public ViewModelCommand SaveCommand => _saveCommand ?? (_saveCommand = new ViewModelCommand(SaveFile));

        private ViewModelCommand _openCommand;
        public ViewModelCommand OpenCommand => _openCommand ?? (_openCommand = new ViewModelCommand(OpenFile));

        // --- Methods ---

        private void SaveFile()
        {
            // Use Livet's Messenger to show File Save Dialog
            var message = new SavingFileMessage("SaveMessage") { Filter = "JSON Files (*.json)|*.json" };
            Messenger.GetResponse(message);

            if (message.Response != null)
            {
                _model.Save(message.Response[0], TextContent);
            }
        }

        private void OpenFile()
        {
            var message = new OpeningFileMessage("OpenMessage") { Filter = "JSON Files (*.json)|*.json" };
            Messenger.GetResponse(message);

            if (message.Response != null)
            {
                var data = _model.Load(message.Response[0]);
                TextContent = data.Content;
            }
        }
    }
